# 股神小易

# 参考程序

```cpp
#include<bits/stdc++.h>
using namespace std;
double a[30][20005];
int ans[30][3];
int main(){
    int n,m;
    double k;
    cin>>n>>m>>k;
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            cin>>a[i][j];
        }
    }
    for(int i=2;i<=n;i++){
        double tmp=1;
        for(int j=1;j<=m;j++){
            if(a[i][j]/a[i-1][j]>tmp){
                ans[i-1][2]=j;
                ans[i][1]=j;
                tmp=a[i][j]/a[i-1][j];
            }
        }
        k*=tmp;
    }
    printf("%.4lf\n",k);
    for(int i=1;i<=n;i++){
        cout<<ans[i][1]-1<<' '<<ans[i][2]-1<<'\n';
    }
    return 0;
}
```
