# 星际探险

## 参考程序

```cpp
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
    ll x,y,z,w,n;
    cin>>x>>y>>z>>w>>n;
    int flag=0;
    ll tmp;
    for(int i=0;i<=2500;i++){
        if(flag) break;
        tmp=n-i*x;
        if(w*2500+z*2500+y*2500<tmp) continue;
        if(tmp<0) break; 
        for(int j=0;j<=2500;j++){
            if(flag) break;
            tmp=n-i*x-j*y;
            if(w*2500+z*2500<tmp) continue;
            if(tmp<0) break; 
            for(int k=0;k<=2500;k++){
                tmp=n-i*x-j*y-k*z;
                if(w*2500<tmp) continue;
                if(tmp<0) break;
                if(w){
                    if(tmp%w==0){
                        cout<<i<<' '<<j<<' '<<k<<' '<<tmp/w<<'\n';
                        flag=1;
                        break;
                    }                    
                }
                else{
                    if(tmp==0) cout<<i<<' '<<j<<' '<<k<<' '<<0<<'\n';
                    flag=1;
                    break;
                }
            }
        }
    }
    if(!flag) cout<<-1;
    return 0;
}
```
